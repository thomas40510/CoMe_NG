def initialize(tokens, semantics = 'ntk')
  @tokens = tokens
  @name = @tokens.find { |token| token.type == ':figName' }.value.first.first
  @figures = []
  semclass = semantics == 'ntk' ? NTKSemantics.new : MelissaSemantics.new
  @sems, @regexes = semclass.sems_and_regexes
end
